 16class ValueConverter(ABC):
 17    """Converts a value from one data representation to another."""
 18
 19    @abstractmethod
 20    def parse_stix_2_1(self, value: Any) -> None:
 21        """Parse a STIX2.1 value.
 22
 23        :param value: The value in STIX 2.1 format
 24        :type value: Any
 25
 26        :raises ValueError: If the value could not be parsed
 27        """
 28        pass
 29
 30    @abstractmethod
 31    def convert_to_typeql(
 32        self,
 33        **kwargs
 34    ) -> Any:
 35        """Create every query or query part that is required to add
 36        this value to the TypeDB database.
 37
 38        :raises ValueError: If the underlying value could not be 
 39            converted
 40        :raises MappingException: If the mapping for the value
 41            transformation process is invalid
 42
 43        :return: The transformed value
 44        :rtype: Any
 45        """
 46        pass
 47
 48    @staticmethod
 49    def get_converter(value_type: str):
 50        """Returns an instance of a ValueConverter for the given type.
 51
 52        :param value_type: The string that specifies the type of the 
 53            converter, e.g. "string" or "long"
 54        :type value_type: str
 55
 56        :raises ValueError: If the value type has no converter 
 57            implementation
 58
 59        :return: An instance of the corresponding ValueConverter
 60        :rtype: ValueConverterAdapter
 61        """
 62        res = None
 63        main_type, subtype = STIXtoTypeDBMapper.split_type(value_type)
 64        match main_type:
 65            case typedb_constants.STRING:
 66                res = StringConverter()
 67            case typedb_constants.DOUBLE:
 68                res = DoubleConverter()
 69            case typedb_constants.BOOLEAN:
 70                res = BooleanConverter()
 71            case typedb_constants.DATETIME:
 72                res = DatetimeConverter()
 73            case typedb_constants.LONG:
 74                res = LongConverter()
 75            case constants.TYPEQL_COMPOSITE_PREFIX:
 76                res = CompositeValueConverter(subtype)
 77            case constants.TYPEQL_RELATION_PREFIX:
 78                res = EmbeddedRelationConverter(subtype)
 79            case constants.TYPEQL_LIST_PREFIX:
 80                res = ListConverter(subtype)
 81            case constants.TYPEQL_NOT_IMPLEMENTED_TYPE_KEYWORD:
 82                res = EmptyValueConverter()
 83            case constants.TYPEQL_DICTIONARY_KEYWORD:
 84                res = DictionaryConverter()
 85            case constants.MAPPING_KEY_VALUE_PAIR_KEYWORD:
 86                res = KeyValuePairConverter(subtype)
 87            case constants.TYPEQL_ROLEPLAYER:
 88                res = EmptyValueConverter()
 89            case _:
 90                raise ValueError(
 91                    f"Value type '{value_type}' has no converter implementation")
 92
 93        return ValueConverterAdapter(res)

 96class ValueConverterAdapter(ValueConverter):
 97    """Adapter class for the ValueConverters.
 98
 99    This class is responsible for giving all ValueConverters the same 
100    return type for their convert_value function.
101    """
102
103    def __init__(self, converter: ValueConverter):
104        """Initializes a Converter Adapter
105
106        :param converter: The underlying converter
107        :type converter: ValueConverter
108        """
109        self.converter = converter
110
111    def parse_stix_2_1(self, value: Any):
112        """Parses a STIX 2.1 value. Delegates this task to the 
113        underlying converter.
114
115        :param value: The value that will be parsed
116        :type value: Any
117        """
118        self.converter.parse_stix_2_1(value)
119
120    def convert_to_typeql(
121        self,
122        **kwargs
123    ) -> tuple[list[str], QueryBundle]:
124        """Converts a value to the TypeDB representation.
125
126        :raises ValueError: If there is a type error concerning 
127            the return types.
128        :raises MappingError: If the underlying mapping from STIX to 
129            the TypeDB schema is invalid
130
131        :return: The values and additional queries required to build 
132            this value in TypeDB
133        :rtype: tuple[list[str], QueryBundle]
134        """
135        res = self.converter.convert_to_typeql(**kwargs)
136        # Primitive types
137        if isinstance(res, str):
138            if not res:
139                return [], QueryBundle()
140            return [res], QueryBundle()
141
142        # List
143        if isinstance(res, tuple):
144            if len(res) != 2:
145                raise ValueError("Value Conversion Type Error")
146            values, queries = res
147            if not isinstance(values, list):
148                raise ValueError("Value Conversion Type Error")
149            if len(values) > 0:
150                if not isinstance(values[0], str):
151                    raise ValueError("Value Conversion Type Error")
152            if not isinstance(queries, QueryBundle):
153                raise ValueError("Value Conversion Type Error")
154            return values, queries
155
156        # Non Primitive types
157        if isinstance(res, QueryBundle):
158            return [], res
159
160        raise ValueError("Value Conversion Type Error")

245class DatetimeConverter(PrimitiveValueConverter):
246    """Converts a *Datetime* value from one representation to another."""
247
248    def __init__(self):
249        self.year: int = 0
250        self.month: int = 0
251        self.day: int = 0
252        self.hour: int = 0
253        self.minute: int = 0
254        self.seconds: int = 0
255        self.subseconds: list[int] = []
256
257    def parse_stix_2_1(self, value):
258        if isinstance(value, STIXdatetime):
259            value = format_datetime(value)
260        if not re.match(stix_constants.STIX_TIMESTAMP_REGEX, value):
261            raise ValueError("Timestamp is not a STIX2.1 timestamp.")
262        values = value.split(stix_constants.TIMESTAMP_DATE_SEPARATOR)
263        self.year = int(values.pop(0))
264        self.month = int(values.pop(0))
265        values = values[0].split(stix_constants.TIMESTAMP_DATE_TIME_SEPARATOR)
266        self.day = int(values.pop(0))
267        values = values[0].split(stix_constants.TIMESTAMP_TIME_SEPARATOR)
268        self.hour = int(values.pop(0))
269        self.minute = int(values.pop(0))
270        if stix_constants.TIMESTAMP_MILLIS_SEPARATOR in values[0]:
271            values = values[0].split(stix_constants.TIMESTAMP_MILLIS_SEPARATOR)
272            self.seconds = int(values.pop(0))
273            millis = values[0] \
274                .replace(stix_constants.TIMESTAMP_ENDING, "")
275            self.subseconds = [int(i) for i in list(millis)]
276        else:
277            self.seconds = int(
278                values[0].replace(stix_constants.TIMESTAMP_ENDING, "")
279            )
280
281    def convert_to_typeql(self, **kwargs):
282        yyyymmdd = str(self.year).zfill(4)
283        yyyymmdd += typedb_constants.DATETIME_DATE_SEPARATOR
284        yyyymmdd += str(self.month).zfill(2)
285        yyyymmdd += typedb_constants.DATETIME_DATE_SEPARATOR
286        yyyymmdd += str(self.day).zfill(2)
287
288        hhmmss = str(self.hour).zfill(2)
289        hhmmss += typedb_constants.DATETIME_TIME_SEPARATOR
290        hhmmss += str(self.minute).zfill(2)
291        hhmmss += typedb_constants.DATETIME_TIME_SEPARATOR
292        hhmmss += str(self.seconds).zfill(2)
293
294        res = yyyymmdd
295        res += typedb_constants.DATETIME_DATE_TIME_SEPARATOR
296        res += hhmmss
297
298        if self.subseconds:
299            millis = self.subseconds
300            num_millis = typedb_constants.DATETIME_NUMBER_MILLIS
301            millis = millis[:num_millis]
302            for _ in range(num_millis - len(millis)):
303                millis.append(0)
304            millis = "".join([str(i) for i in millis])
305            millis = millis.zfill(num_millis)
306            res += typedb_constants.DATETIME_MILLIS_SEPARATOR
307            res += millis
308        return res

326class KeyValuePairConverter(NonPrimitiveValueConverter):
327    """Converts predefined KeyValuePairs from one representation 
328    to another.
329    """
330
331    def __init__(self, subtype: str):
332        super().__init__(subtype)
333        self.key_value_pairs: dict = dict()
334
335    def parse_stix_2_1(self, value: dict):
336        self.key_value_pairs = value
337
338    def convert_to_typeql(self, **kwargs):
339        queries = QueryBundle()
340
341        value_attribute, value_type = STIXtoTypeDBMapper \
342            .get_pairs_value_and_type(self.typedb_subtype)
343
344        for key, value in self.key_value_pairs.items():
345            entity = Entity()
346
347            # Transform the key
348            typedb_type = STIXtoTypeDBMapper \
349                .get_key_value_pair_key_translation(self.typedb_subtype, key)
350
351            # Custom key
352            if typedb_type is None:
353                typedb_type, key_attribute = STIXtoTypeDBMapper \
354                    .get_key_value_pair_custom(self.typedb_subtype)
355                key = typedb_constants.to_typedb_string(key)
356                entity.add_attribute(key_attribute, key)
357
358            entity.set_type(typedb_type)
359
360            # Transform the value
361            # Raises ValueError
362            converter: ValueConverter = ValueConverter \
363                .get_converter(value_type)
364            converter.parse_stix_2_1(value)
365            values, additional_queries = converter.convert_to_typeql()
366            if not additional_queries.is_empty():
367                raise ValueError(
368                    "NonPrimitive Types are currently not supported for "
369                    "key value pairs"
370                )
371            for value in values:
372                entity.add_attribute(value_attribute, value)
373
374            # build the relation
375            relation_name, object_role, item_role = STIXtoTypeDBMapper \
376                .get_pair_relation(self.typedb_subtype)
377            relation = Relation(typedb_type=relation_name)
378
379            reference: Identification = kwargs \
380                .get(constants.REFERENCE_KEYWORD)
381            if not reference:
382                raise ValueError(
383                    "{constants.REFERENCE_KEYWORD} parameter missing in function call "
384                    "for key-value pair creation")
385            relation.add_roleplayer(object_role, reference.get_variable())
386            relation.add_roleplayer(item_role, entity.get_variable())
387
388            queries.add_structured_attribute(entity, relation)
389
390        return queries

393class CompositeValueConverter(NonPrimitiveValueConverter):
394    """Converts a composite value from one representation to another."""
395
396    def __init__(self, subtype: str):
397        super().__init__(subtype)
398        self.attributes = dict()
399
400    def parse_stix_2_1(self, value: dict):
401        self.attributes: dict = value
402
403    def convert_to_typeql(self, **kwargs):
404        # Set up the entity that represent the composite type
405        queries = QueryBundle()
406        entity = Entity(typedb_type=self.typedb_subtype)
407
408        # set up the attributes of the composite type
409        selfref = Identification(
410            "",
411            entity.get_variable(),
412            self.typedb_subtype
413        )
414        for stix_name, stix_value in self.attributes.items():
415            # Get the attribute name and value type in TypeDB
416            typedb_name, typedb_type = STIXtoTypeDBMapper \
417                .get_composite_attribute_info(self.typedb_subtype, stix_name)
418
419            # Convert the value
420            converter: ValueConverter = ValueConverter \
421                .get_converter(typedb_type)
422            converter.parse_stix_2_1(stix_value)
423            typedb_values, additional_queries = converter \
424                .convert_to_typeql(reference=selfref)
425
426            for typedb_value in typedb_values:
427                entity.add_attribute(typedb_name, typedb_value)
428            queries.extend(additional_queries)
429
430        reference: Identification = kwargs.get(constants.REFERENCE_KEYWORD)
431        if not reference:
432            raise ValueError(
433                "{constants.REFERENCE_KEYWORD} parameter not specified for conversion of composite type")
434
435        # set up the helper relation
436        relation_name = STIXtoTypeDBMapper \
437            .get_composite_type_relation(self.typedb_subtype)
438        relation = Relation(typedb_type=relation_name)
439        object_role, value_role = STIXtoTypeDBMapper \
440            .get_composite_relation_roles(self.typedb_subtype)
441        relation.add_roleplayer(object_role, reference.get_variable())
442        relation.add_roleplayer(value_role, selfref.get_variable())
443
444        queries.add_structured_attribute(entity, relation)
445        return queries

448class EmbeddedRelationConverter(NonPrimitiveValueConverter):
449    """Converts a relation from one representation to another."""
450
451    def __init__(self, subtype: str):
452        super().__init__(subtype)
453
454    def parse_stix_2_1(self, value):
455        self.target_id: str = value
456
457    def get_id_and_roles(self, **kwargs) -> tuple[Identification, str, str]:
458        """Extracts the id of the calling object and the name of the 
459        object's and value's roles.
460
461        :raises ValueError: If a required parameter is not given
462
463        :return: The id of the object and roles of the object and value
464        :rtype: tuple[str, str, str]
465        """
466        reference: Identification = kwargs.get(constants.REFERENCE_KEYWORD)
467        if not reference:
468            raise ValueError(
469                f"Parameter '{constants.REFERENCE_KEYWORD}' not given for embedded relation conversion")
470        attribute_name = kwargs.get(constants.ATTRIBUTE_NAME)
471        if not attribute_name:
472            raise ValueError(
473                f"Parameter '{constants.ATTRIBUTE_NAME}' not given for embedded relation conversion")
474        obj_role, val_role = STIXtoTypeDBMapper \
475            .get_embedded_relation_roles(attribute_name)
476        return reference, obj_role, val_role
477
478    def convert_to_typeql(self, **kwargs):
479        # Get required information
480        relation_name = self.typedb_subtype
481        reference, object_role, value_role = self.get_id_and_roles(**kwargs)
482        value_id = self.target_id
483
484        try:
485            stix_value_type = get_type_from_id(value_id)
486        except AttributeError as err:
487            raise ValueError(f"Error getting type of STIX object with id '{value_id}':\n{err}") from err
488        value_type = STIXtoTypeDBMapper.get_typedb_supertype(stix_value_type)
489
490        # set up the entities
491        value_entity = Entity(typedb_type=value_type)
492        value_id = typedb_constants.to_typedb_string(value_id)
493        value_entity.add_attribute(constants.TYPEDB_ID_ATTRIBUTE, value_id)
494        object_entity = reference.get_match_object()
495
496        # set up the relation
497        relation = Relation(typedb_type=relation_name)
498        relation.add_roleplayer(object_role, object_entity.get_variable())
499        relation.add_roleplayer(value_role, value_entity.get_variable())
500
501        queries = QueryBundle()
502        queries.add_embedded_relation(object_entity, value_entity, relation)
503
504        return queries
