 21    @staticmethod
 22    def create(properties: dict) -> Self:
 23        """Creates the corresponding subclass according 
 24        to the properties.
 25
 26        :param properties: The properties of the STIX Object
 27        :type properties: dict
 28
 29        :raises TransformationError: If the 'id' or 'type' property is 
 30            missing
 31        :raises MappingException: If the STIX object class is not 
 32            defined
 33
 34        :return: The STIX Object (as a subclass instance)
 35        :rtype: STIXObject
 36        """
 37        logger.debug(log_messages.CREATE_START)
 38
 39        stix_id = properties.get(stix_constants.STIX_PROPERTY_ID)
 40        if not stix_id:
 41            raise TransformationError(TransformationError.MISSING_REQUIRED_PROPERTY, "'id'")
 42
 43        stix_type = properties.get(stix_constants.STIX_PROPERTY_TYPE)
 44        if not stix_type:
 45            raise TransformationError(TransformationError.MISSING_REQUIRED_PROPERTY, "\"type\"", stix_id)
 46
 47        class_type = STIXtoTypeDBMapper.get_class(stix_type)
 48
 49        match class_type:
 50            case constants.MAPPING_CLASS_SDO:
 51                obj = STIXDomainObjectConverter(properties)
 52            case constants.MAPPING_CLASS_SRO:
 53                obj = STIXRelationshipObjectConverter(properties)
 54            case constants.MAPPING_CLASS_SCO:
 55                obj = STIXCyberObservableConverter(properties)
 56            case constants.MAPPING_CLASS_SMO:
 57                obj = STIXMetaObjectConverter(properties)
 58            case None:
 59                obj = STIXCustomObjectConverter(properties)
 60            case _:
 61                raise MappingException(
 62                    f"Undefined STIX class keyword: {class_type}")
 63
 64        logger.debug(log_messages.CREATE_SUCCESS(stix_id, stix_type, class_type))
 65
 66        return obj

186    def get_relevant_extension(self) -> str:
187        """Determine which extension to choose for converting.
188
189        :return: The name of the relevant extension
190        :rtype: str
191        """
192        logger.debug(log_messages.EXTENSIONS_FIND_START)
193
194        extension = ""
195        for ext in self.extensions.keys():
196            try:
197                is_defined = STIXtoTypeDBMapper.is_defined_extension(
198                    self.stix_type, ext
199                )
200            except MappingException as e:
201                logger.error(
202                    log_messages.MAPPING_INVALID.format(
203                        reference=self.stix_id,
204                        stix_name=ext,
205                        stix_value="",
206                        exception=e
207                    )
208                )
209                continue
210
211            if is_defined:
212                if extension:
213                    logger.warning(
214                        log_messages.EXTENSIONS_FIND_MULTIPLE_DEFINED, ext)
215                else:
216                    extension = ext
217                    logger.debug(
218                        log_messages.EXTENSIONS_FIND_FOUND, ext)
219            else:
220                logger.warning(
221                    log_messages.EXTENSIONS_FIND_CUSTOM, ext)
222        if extension:
223            logger.debug(
224                log_messages.EXTENSIONS_FIND_FINAL, extension)
225        else:
226            logger.debug(log_messages.EXTENSIONS_FIND_NOTHING_FOUND)
227        return extension

522class STIXCoreObjectConverter(STIXObjectConverter):
523    """Represents STIX Core objects."""
524
525    @abstractmethod
526    def create_typeql_object(
527        self,
528        type_name: str,
529        queries: QueryBundle
530    ) -> tuple[Thing, Identification]:
531        """Create the entity or relation that represents the main object 
532        and add it to the builder.
533
534        :param type_name: The TypeDB type of the main object
535        :type type_name: str
536        :param queries: The QueryBundle to which the created object will
537            be added
538        :type queries: QueryBundle
539
540        :raises TransformationError: If the STIX Object does not have 
541            required properties
542        :raises MappingException: If the mapping is invalid
543
544        :return: The thing that represents the main STIX Object and 
545            a reference to it 
546        :rtype: tuple[Thing, Identification]
547        """
548
549    def build_typeql_bundle(self):
550        logger.debug(log_messages.BUILD_CORE_START,self.stix_id)
551
552        # queries needed to set up this object
553        queries = QueryBundle()
554
555        # raises MappingException and TransformationError
556        type_name = self.determine_typedb_name()
557        if not type_name:
558            raise TransformationError(TransformationError.UNSUPPORTED_CUSTOM_TYPE, reference_id=self.stix_id)
559
560        thing, ref = self.create_typeql_object(type_name, queries)
561
562        # transform the attributes
563        attributes, additional_queries = self.create_typeql_properties(ref)
564        queries.extend(additional_queries)
565        self.add_attributes_to_typedb_object(attributes, thing)
566
567        # transform the extensions
568        attributes, additional_queries = self.create_typeql_extensions(ref)
569        queries.extend(additional_queries)
570        self.add_attributes_to_typedb_object(attributes, thing)
571
572        return queries

603class STIXRelationshipObjectConverter(STIXCoreObjectConverter):

606    def create_role(
607        self,
608        role_name: str,
609        roleplayer_id: str,
610        relation: Relation
611    ) -> Entity:
612        """Create a single role for this relation.
613
614        :param role_name: The name of the role
615        :type role_name: str
616        :param roleplayer_id: The id of the roleplayer
617        :type roleplayer_id: str
618        :param relation: The main object relation
619        :type relation: Relation
620
621        :return: The roleplayer's match object
622        :rtype: Entity
623        """
624        stix_type = get_type_from_id(roleplayer_id)
625        object_type = STIXtoTypeDBMapper.get_typedb_supertype(stix_type)
626
627        match_object = Identification(roleplayer_id, "", object_type) \
628            .get_match_object()
629        relation.add_roleplayer(role_name, match_object.get_variable())
630        return match_object

632    def create_role_property(
633        self,
634        type_name: str,
635        property_name: str,
636        relation: Relation
637    ) -> list[Entity]:
638        """Creates a role property of the main relation, i.e. creates 
639        the match objects and returns them.
640
641        Note: Because a property's value type might be a list of 
642            identifiers, we have to return a list of match objects
643
644        :param type_name: The TypeDB type for this object
645        :type type_name: str
646        :param property_name: The name of the property that is 
647            implemented as a role player
648        :type property_name: str
649        :param relation: The main relation object
650        :type relation: Relation
651
652        :raises TransformationError: If the property does not exist
653        :raises MappingException:
654
655        :return: The match objects for the created roleplayers
656        :rtype: list[Entity]
657        """
658        match_objects = []
659
660        role_name = STIXtoTypeDBMapper.get_sro_role(type_name, property_name)
661        roleplayer_id = self.properties.get(property_name)
662
663        if not roleplayer_id:
664            raise TransformationError(TransformationError.UNSPECIFIED_PROPERTY, property_name, self.stix_id)
665        if isinstance(roleplayer_id, str):
666            match_objects = [
667                self.create_role(role_name, roleplayer_id, relation)
668            ]
669        elif isinstance(roleplayer_id, list):
670            roleplayer_ids = roleplayer_id
671            if roleplayer_ids and isinstance(roleplayer_ids[0], str):
672                for roleplayer_id in roleplayer_ids:
673                    match_objects.append(
674                        self.create_role(role_name, roleplayer_id, relation)
675                    )
676            else:
677                raise TransformationError(
678                    TransformationError.INVALID_PROPERTY_VALUE, property, self.stix_id)
679        else:
680            raise TransformationError(
681                TransformationError.INVALID_PROPERTY_TYPE, property, self.stix_id)
682
683        return match_objects

685    def create_typeql_object(self, type_name, queries):
686        relation = Relation(typedb_type=type_name)
687        ref = Identification(
688            self.get_stix_id(),
689            relation.get_variable(),
690            type_name
691        )
692
693        # create roles
694        relation_properties = STIXtoTypeDBMapper \
695            .get_roleplayers_for_SRO(self.stix_type)
696        match_objects = []
697        for property in relation_properties:
698            if property in self.properties:
699                # raises MappingException and TransformationError
700                role_players = self.create_role_property(
